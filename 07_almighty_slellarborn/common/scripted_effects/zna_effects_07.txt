# Borders (Solar system scope)
create_almighty_starbases = {
	if = {
		limit = {
			solar_system = {
				NOR = {
					any_ship_in_system = {
						owner = { is_hostile = event_target:sofe_crisis_country }
					}
					any_system_planet = {
						NOR = {
							merg_is_void_system = yes
							merg_is_empty_space = yes
						}
						exists = owner
						owner = {
							is_hostile = event_target:sofe_crisis_country
						}
					}
				}
			}
		}
		create_almighty_citadel = yes
	}
}

	create_almighty_leader = yes
	create_fleet = {
		name = "NAME_ALMIGHTY_SCOUTS"
		settings = { 
			can_upgrade = no
			can_disband = no
			uses_naval_capacity = no
			is_boss = yes
			spawn_debris = no
			can_change_composition = no 
			can_change_leader = no 
		}
		effect = {
			set_owner = event_target:sofe_crisis_country
			if = {
				limit = { has_global_flag = almighty_nani }
				while = {
					count = 16
					create_ship = {
						name = random
						design = "NAME_ALMIGHTY_ESCORT"
					}
				}
			} else = {
				while = {
					count = 3
					create_ship = {
						name = random
						design = "NAME_ALMIGHTY_ESCORT"
					}
				}
			}
			assign_leader = last_created_leader
			set_fleet_flag = sofe_almighty_scouts
			set_formation_scale = 2
			set_fleet_stance = aggressive
			set_aggro_range_measure_from = self
			set_aggro_range = 250
			set_location = {
				target = prev
				distance = 0
				angle = 0
			}
		}
	}
}
